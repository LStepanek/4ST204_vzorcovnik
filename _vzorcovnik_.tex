\subsection{Četnosti}
%%-----------------------------------------------------------------------------

\subsubsection{Absolutní četnost}

Absolutní četnost $i$-té hodnoty je počet výskytů dané hodnoty ve výběru.

\begin{equation*}
n_i
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $n_i$ je počet výskytů $i$-té hodnoty ve výběru.
\end{itemize}
%%-----------------------------------------------------------------------------

\subsubsection{Součet absolutních četností}

Součet absolutních četností všech hodnot (nebo tříd) ve výběru
se rovná celkovému počtu pozorování.

\begin{equation*}
\sum_{i=1}^{k} n_i = n
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\sum$ je sumační znak (značí sčítání),
\item $i$ je index hodnoty nebo třídy,
\item $k$ je počet různých hodnot nebo tříd,
\item $n_i$ je absolutní četnost $i$-té hodnoty (nebo třídy),
\item $n$ je celkový počet pozorování.
\end{itemize}
%%-----------------------------------------------------------------------------

\subsubsection{Relativní Četnost}
Relativní četnost $i$-té hodnoty ve výběru je definována jako podíl
absolutní četnosti dané hodnoty a celkového počtu pozorování,

\begin{equation*}
p_i = \frac{n_i}{n}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $p_i$ je relativní četnost $i$-té hodnoty,
\item $n_i$ je absolutní četnost $i$-té hodnoty (kolikrát se hodnota vyskytla),
\item $n$ je celkový počet pozorování v souboru.
\end{itemize}
%%-----------------------------------------------------------------------------

\subsubsection{Součet relativních četností}

Součet relativních četností všech hodnot (nebo tříd) ve výběru
je roven jedné.

\begin{equation*}
\sum_{i=1}^{k} p_i = 1
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\sum$ je sumační znak (značí sčítání),
\item $i$ je index hodnoty nebo třídy,
\item $k$ je počet různých hodnot nebo tříd,
\item $p_i$ je relativní četnost $i$-té hodnoty (nebo třídy).
\end{itemize}

%%-----------------------------------------------------------------------------
\subsection{Charakteristiky polohy}
%%_____________________________________________________________________________

\subsubsection{Aritmetický průměr}
%%-----------------------------------------------------------------------------

Aritmetický průměr (výběrový průměr) je definován jako součet
všech hodnot statistického souboru dělený počtem pozorování.

\begin{equation*}
\bar{x} = \frac{\sum_{i=1}^{n} x_i}{n}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\bar{x}$ je aritmetický průměr,
\item $\sum$ je sumační znak (součet hodnot),
\item $x_i$ je $i$-tá hodnota statistického souboru,
\item $i$ je index hodnoty,
\item $n$ je počet pozorování.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{mean(x)} &
\begin{lstlisting}
x <- c(1,2,3,4,5)
mean(x)
\end{lstlisting}
&
Vrací aritmetický průměr hodnot ve vektoru \texttt{x}.
\\
\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.mean(x)} &
\begin{lstlisting}
import numpy as np
x = np.array([1,2,3,4,5])
np.mean(x)
\end{lstlisting}
&
Vrací aritmetický průměr pole pomocí knihovny \texttt{numpy}.
\\
\end{funkcedict}

%%----------------------------------------------------------------------------
\subsubsection{Vážený aritmetický průměr (z četnostní tabulky)}

Pokud jsou data zadána pomocí četnostní tabulky, aritmetický průměr
lze vypočítat jako vážený průměr hodnot.

\begin{equation*}
\bar{x} =
\frac{\sum_{i=1}^{k} x_i n_i}{\sum_{i=1}^{k} n_i}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\bar{x}$ je aritmetický průměr,
\item $\sum$ je sumační znak (součet),
\item $x_i$ je $i$-tá hodnota znaku,
\item $n_i$ je absolutní četnost $i$-té hodnoty,
\item $k$ je počet různých hodnot nebo tříd,
\item $\sum_{i=1}^{k} n_i$ je celkový počet pozorování.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{weighted.mean(x, w)} &
\begin{lstlisting}
x <- c(1,2,3)
w <- c(2,3,5)

weighted.mean(x, w)
\end{lstlisting}
&
Vrací vážený průměr hodnot ve vektoru \texttt{x} s váhami \texttt{w}.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.average(x, weights=w)} &
\begin{lstlisting}
import numpy as np

x = np.array([1,2,3])
w = np.array([2,3,5])

np.average(x, weights=w)
\end{lstlisting}
&
Vrací vážený průměr hodnot ve vektoru \texttt{x} s váhami \texttt{w}.
\\

\end{funkcedict}

%%-----------------------------------------------------------------------------
\subsubsection{Aritmetický průměr pomocí relativních četností}

Pokud jsou data vyjádřena pomocí relativních četností,
lze aritmetický průměr vypočítat jako vážený součet hodnot.

\begin{equation*}
\bar{x} = \sum_{i=1}^{k} x_i p_i
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\bar{x}$ je aritmetický průměr,
\item $\sum$ je sumační znak (součet),
\item $x_i$ je $i$-tá hodnota znaku,
\item $p_i$ je relativní četnost $i$-té hodnoty,
\item $i$ je index hodnoty,
\item $k$ je počet různých hodnot nebo tříd.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{sum(x * p)} &
\begin{lstlisting}
x <- c(1,2,3)
p <- c(0.2,0.3,0.5)

sum(x * p)
\end{lstlisting}
&
Vrací aritmetický průměr vypočtený pomocí relativních četností.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.sum(x * p)} &
\begin{lstlisting}
import numpy as np

x = np.array([1,2,3])
p = np.array([0.2,0.3,0.5])

np.sum(x * p)
\end{lstlisting}
&
Vrací aritmetický průměr vypočtený pomocí relativních četností
pomocí knihovny \texttt{numpy}.
\\

\end{funkcedict}

%%-----------------------------------------------------------------------------
\subsubsection{Harmonický průměr}

Harmonický průměr je definován jako podíl počtu pozorování
a součtu převrácených hodnot jednotlivých pozorování.

\begin{equation*}
\bar{x}_H = \frac{n}{\sum_{i=1}^{n} \frac{1}{x_i}}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\bar{x}_H$ je harmonický průměr,
\item $n$ je počet pozorování,
\item $\sum$ je sumační znak (součet),
\item $x_i$ je $i$-tá hodnota statistického souboru,
\item $\frac{1}{x_i}$ je převrácená hodnota $i$-tého pozorování.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{n / sum(1/x)} &
\begin{lstlisting}
x <- c(1,2,4,5)
n <- length(x)

n / sum(1/x)
\end{lstlisting}
&
Vrací harmonický průměr hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{scipy.stats. hmean(x)} &
\begin{lstlisting}
from scipy.stats import hmean

x = [1,2,4,5]

hmean(x)
\end{lstlisting}
&
Vrací harmonický průměr pomocí funkce \texttt{hmean}
z knihovny \texttt{scipy.stats}.
\\

\end{funkcedict}

%%-----------------------------------------------------------------------------
\subsubsection{Geometrický průměr}

Geometrický průměr je definován jako $n$-tá odmocnina součinu
všech hodnot statistického souboru.

\begin{equation*}
\bar{x}_G =
\sqrt[n]{\prod_{i=1}^{n} x_i}
=
\sqrt[n]{x_1 \cdot x_2 \cdots x_n}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\bar{x}_G$ je geometrický průměr,
\item $\prod$ je znak pro součin (násobení všech hodnot),
\item $x_i$ je $i$-tá hodnota statistického souboru,
\item $i$ je index hodnoty,
\item $n$ je počet pozorování,
\item $\sqrt[n]{\cdot}$ značí $n$-tou odmocninu.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{exp(mean(log(x)))} &
\begin{lstlisting}
x <- c(1,2,3,4)

exp(mean(log(x)))
\end{lstlisting}
&
Vrací geometrický průměr hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{scipy.stats. gmean(x)} &
\begin{lstlisting}
from scipy.stats import gmean

x = [1,2,3,4]

gmean(x)
\end{lstlisting}
&
Vrací geometrický průměr pomocí funkce \texttt{gmean}
z knihovny \texttt{scipy.stats}.
\\

\end{funkcedict}

%%-----------------------------------------------------------------------------
\subsubsection{Vztah mezi průměry}

\begin{equation*}
\bar{x}_H \le \bar{x}_G \le \bar{x}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\bar{x}_H$ je harmonický průměr,
\item $\bar{x}_G$ je geometrický průměr,
\item $\bar{x}$ je aritmetický průměr.
\end{itemize}

\subsubsection{Medián}

Medián je hodnota, která rozděluje uspořádaný statistický soubor
na dvě stejně početné části. Medián také vyjadřuje 50\% kvantil.

\begin{equation*}
\tilde{x} =
\begin{cases}
x_{\left(\frac{n+1}{2}\right)}, & \text{je-li } n \text{ liché}, \\
\\
\frac{x_{\left(\frac{n}{2}\right)} + x_{\left(\frac{n}{2}+1\right)}}{2}, & \text{je-li } n \text{ sudé}.
\end{cases}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\tilde{x}$ je medián,
\item $x_{(i)}$ je $i$-tá hodnota v uspořádaném statistickém souboru,
\item $n$ je počet pozorování.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{median(x)} &
\begin{lstlisting}
x <- c(1,3,2,5,4)
median(x)
\end{lstlisting}
&
Vrací medián hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.median(x)} &
\begin{lstlisting}
import numpy as np

x = np.array([1,3,2,5,4])
np.median(x)
\end{lstlisting}
&
Vrací medián hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}

%%----------------------------------------------------------------------------
\subsubsection{Kvantily a kvartily}

Kvantily rozdělují uspořádaný statistický soubor na části
se stejným počtem pozorování.

$q$-kvantil je hodnota $x_q$, pro kterou platí

\begin{equation*}
P(X \le x_q) = q
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $x_q$ je $q$-kvantil,
\item $q \in (0,1)$ je pořadí kvantilu,
\item $P(X \le x_q)$ je pravděpodobnost, že náhodná veličina
nabývá hodnot menších nebo rovných $x_q$.
\end{itemize}

Speciálním případem kvantilů jsou kvartily, které rozdělují
data na čtyři stejně velké části.

\begin{equation*}
Q_1 = x_{0.25}, \qquad
Q_2 = x_{0.5}, \qquad
Q_3 = x_{0.75}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $Q_1$ je první kvartil (25\% dat je menších),
\item $Q_2$ je druhý kvartil, který odpovídá mediánu,
\item $Q_3$ je třetí kvartil (75\% dat je menších).
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{quantile(x)} &
\begin{lstlisting}
x <- c(1,2,3,4,5,6,7,8)

quantile(x)
\end{lstlisting}
&
Vrací kvantily datového vektoru.
\\

\texttt{quantile(x, probs=c(0.25, 0.5, 0.75))} &
\begin{lstlisting}
x <- c(1,2,3,4,5,6,7,8)

quantile(x, probs=c(0.25,0.5,0.75))
\end{lstlisting}
&
Vrací první, druhý a třetí kvartil.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.quantile(x,q)} &
\begin{lstlisting}
import numpy as np

x = np.array([1,2,3,4,5,6,7,8])

np.quantile(x,0.5)
\end{lstlisting}
&
Vrací $q$-kvantil pole.
\\

\texttt{np.quantile(x, [0.25,0.5,0.75])} &
\begin{lstlisting}
import numpy as np

x = np.array([1,2,3,4,5,6,7,8])

np.quantile(x,[0.25,0.5,0.75])
\end{lstlisting}
&
Vrací první, druhý a třetí kvartil.
\\

\end{funkcedict}
%%----------------------------------------------------------------------------
\subsubsection{Interkvartilové rozpětí (IQR)}

Interkvartilové rozpětí (IQR) je rozdíl mezi třetím
a prvním kvartilem. Vyjadřuje rozptyl prostředních
50\,\% dat.

\begin{equation*}
IQR = Q_3 - Q_1
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $IQR$ je interkvartilové rozpětí,
\item $Q_1$ je první kvartil,
\item $Q_3$ je třetí kvartil.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{IQR(x)} &
\begin{lstlisting}
x <- c(1,2,3,4,5,6,7,8)

IQR(x)
\end{lstlisting}
&
Vrací interkvartilové rozpětí datového vektoru.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.quantile(x,0.75) - np.quantile(x,0.25)} &
\begin{lstlisting}
import numpy as np

x = np.array([1,2,3,4,5,6,7,8])

np.quantile(x,0.75) - np.quantile(x,0.25)
\end{lstlisting}
&
Vrací interkvartilové rozpětí pole.
\\

\end{funkcedict}

\subsubsection{Variační rozpětí}

Variační rozpětí je rozdíl mezi maximální a minimální
hodnotou statistického souboru.

\begin{equation*}
R = x_{\max} - x_{\min}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $R$ je variační rozpětí,
\item $x_{\max}$ je maximální hodnota v souboru,
\item $x_{\min}$ je minimální hodnota v souboru.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{max(x) - min(x)} &
\begin{lstlisting}
x <- c(3,7,2,9,5)

max(x) - min(x)
\end{lstlisting}
&
Vrací variační rozpětí hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}

\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.max(x) - np.min(x)} &
\begin{lstlisting}
import numpy as np

x = np.array([3,7,2,9,5])

np.max(x) - np.min(x)
\end{lstlisting}
&
Vrací variační rozpětí hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}

%%-----------------------------------------------------------------------------
\subsubsection{Rozptyl statistického souboru}

Rozptyl vyjadřuje průměrnou kvadratickou odchylku hodnot
od aritmetického průměru.

\begin{equation*}
s_x^2 = \frac{\sum_{i=1}^{n} (x_i - \bar{x})^2}{n}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $s_x^2$ je rozptyl statistického souboru,
\item $\sum$ je sumační znak (součet),
\item $x_i$ je $i$-tá hodnota statistického souboru,
\item $\bar{x}$ je aritmetický průměr,
\item $x_i - \bar{x}$ je odchylka hodnoty od průměru,
\item $(x_i - \bar{x})^2$ je čtverec odchylky,
\item $n$ je počet pozorování.
\end{itemize}

Výpočetní tvar rozptylu:

\begin{equation*}
s_x^2 =
\frac{\sum_{i=1}^{n} x_i^2}{n}
-
\left(
\frac{\sum_{i=1}^{n} x_i}{n}
\right)^2
\end{equation*}


\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{mean((x - mean(x))$^2$)} &
\begin{lstlisting}
x <- c(2,4,6,8)

mean((x-mean(x))^2)
\end{lstlisting}
&
Vrací nevýběrový (populační) rozptyl
hodnot ve vektoru \texttt{x}.
\\

\end{funkcedict}


\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\texttt{np.var(x)} &
\begin{lstlisting}
import numpy as np

x = np.array([2,4,6,8])

np.var(x)
\end{lstlisting}
&
Vrací nevýběrový (populační) rozptyl
pomocí knihovny \texttt{numpy}.
\\

\end{funkcedict}

%%----------------------------------------------------------------------------
\subsubsection{Výběrový rozptyl}

Výběrový rozptyl je nezkreslený odhad rozptylu populace
a vznikne úpravou rozptylu statistického souboru pomocí
Besselovy korekce.

\begin{equation*}
S^2 = \frac{n}{n-1}s_x^2
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $S^2$ je výběrový rozptyl,
\item $s_x^2$ je rozptyl statistického souboru,
\item $n$ je počet pozorování,
\item $n-1$ je korekce při odhadu rozptylu (Besselova korekce).
\end{itemize}


\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\texttt{var(x)} &
\begin{lstlisting}
x <- c(2,4,6,8)

var(x)
\end{lstlisting}
&
Vrací výběrový rozptyl hodnot ve vektoru
\texttt{x} (dělí se $n-1$).
\\

\end{funkcedict}


\paragraph{Implementace v Python}\mbox{}\par

\begin{funkcedict}

\begin{lstlisting}

np.var(x, ddof=1)
\end{lstlisting}
&
Vrací výběrový rozptyl pomocí knihovny
\texttt{numpy} (Besselova korekce).
\\

\end{funkcedict}

%%-----------------------------------------------------------------------------
\subsubsection{Vážený rozptyl}

Vážený rozptyl vyjadřuje variabilitu hodnot znaku,
když mají jednotlivé hodnoty různé váhy nebo relativní četnosti.

\begin{equation*}
s^2 = \sum_{i=1}^{k} p_i (x_i - \bar{x})^2
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $s^2$ je vážený rozptyl,
\item $x_i$ je $i$-tá hodnota znaku,
\item $\bar{x}$ je vážený aritmetický průměr,
\item $p_i$ je váha nebo relativní četnost $i$-té hodnoty,
\item $k$ je počet různých hodnot nebo tříd.
\end{itemize}

\paragraph{Implementace v R}\mbox{}\par

\begin{funkcedict}

\end{funkcedict}

\end{funkcedict}

\section{Teorie pravděpodobnosti}
%%_____________________________________________________________________________
\subsection{Počet pravděpodobnosti}
%%-----------------------------------------------------------------------------
\subsubsection{Klasická definice pravděpodobnosti}

Pravděpodobnost jevu $A$ při klasické definici pravděpodobnosti
je podíl počtu příznivých výsledků a celkového počtu
stejně pravděpodobných výsledků.

\begin{equation*}
P(A) = \frac{m}{n}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(A)$ je pravděpodobnost jevu $A$,
\item $m$ je počet příznivých výsledků,
\item $n$ je počet všech možných výsledků.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Pravděpodobnost komplementárního jevu}

Pravděpodobnost opačného (komplementárního, doplňkového) jevu k jevu $A$
je rovna jedné minus pravděpodobnost jevu $A$.

\begin{equation*}
P(\overline{A}) = 1 - P(A)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(\overline{A})$ je pravděpodobnost opačného jevu k $A$,
\item $P(A)$ je pravděpodobnost jevu $A$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Horní odhad pravděpodobnosti průniku}

Pravděpodobnost současného nastání dvou jevů nemůže být větší
než pravděpodobnost kteréhokoli z nich.

\begin{equation*}
P(A \cap B) \le \min(P(A), P(B))
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(A \cap B)$ je pravděpodobnost, že nastanou oba jevy současně,
\item $P(A)$ je pravděpodobnost jevu $A$,
\item $P(B)$ je pravděpodobnost jevu $B$,
\item $\min(\cdot)$ označuje minimum z daných hodnot.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Průnik nezávislých jevů}

Jsou-li jevy $A$ a $B$ nezávislé, pak pravděpodobnost jejich
současného nastání je rovna součinu jejich pravděpodobností.

\begin{equation*}
P(A \cap B) = P(A)P(B)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(A \cap B)$ je pravděpodobnost současného nastání jevů $A$ a $B$,
\item $P(A)$ je pravděpodobnost jevu $A$,
\item $P(B)$ je pravděpodobnost jevu $B$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Sjednocení dvou jevů}

Pravděpodobnost, že nastane alespoň jeden z jevů $A$ nebo $B$,
je rovna součtu jejich pravděpodobností minus pravděpodobnost
jejich současného nastání.

\begin{equation*}
P(A \cup B) = P(A) + P(B) - P(A \cap B)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(A \cup B)$ je pravděpodobnost, že nastane alespoň jeden z jevů,
\item $P(A)$ je pravděpodobnost jevu $A$,
\item $P(B)$ je pravděpodobnost jevu $B$,
\item $P(A \cap B)$ je pravděpodobnost současného nastání jevů $A$ a $B$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Sjednocení neslučitelných jevů}

Pokud jsou jevy $A$ a $B$ neslučitelné (nemohou nastat současně),
pak pravděpodobnost jejich sjednocení je rovna součtu
pravděpodobností těchto jevů.

\begin{equation*}
P(A \cup B) = P(A) + P(B)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(A \cup B)$ je pravděpodobnost, že nastane jev $A$ nebo $B$,
\item $P(A)$ je pravděpodobnost jevu $A$,
\item $P(B)$ je pravděpodobnost jevu $B$.
\end{itemize}

%%_____________________________________________________________________________
\subsection{Náhodné veličiny}
%%-----------------------------------------------------------------------------
\subsubsection{Pravděpodobnostní funkce diskrétní náhodné veličiny}

Pravděpodobnost, že diskrétní náhodná veličina $X$
nabývá konkrétní hodnoty $x$.

\begin{equation*}
P(x) = P(X = x)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(x)$ je pravděpodobnost hodnoty $x$,
\item $P(X = x)$ je pravděpodobnost, že náhodná veličina $X$ má hodnotu $x$,
\item $X$ je náhodná veličina,
\item $x$ je konkrétní hodnota náhodné veličiny.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Distribuční funkce diskrétní náhodné veličiny}

Distribuční funkce (kumulativní pravděpodobnost) udává
pravděpodobnost, že náhodná veličina $X$ bude menší
nebo rovna hodnotě $x$.

\begin{equation*}
F(x) = P(X \le x) = \sum_{x_j \le x} P(x_j)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $F(x)$ je distribuční funkce,
\item $P(X \le x)$ je pravděpodobnost, že $X$ je menší nebo rovno $x$,
\item $\sum$ je součet pravděpodobností,
\item $x_j$ jsou jednotlivé možné hodnoty náhodné veličiny,
\item $P(x_j)$ je pravděpodobnost, že $X = x_j$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Pravděpodobnost intervalu}

Pravděpodobnost, že náhodná veličina $X$ nabývá hodnot
v intervalu $(x_1, x_2]$.

\begin{equation*}
P(x_1 < X \le x_2)
=
\sum_{x_1 < x \le x_2} P(x)
=
F(x_2) - F(x_1)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(x_1 < X \le x_2)$ je pravděpodobnost, že $X$ leží mezi $x_1$ a $x_2$,
\item $X$ je náhodná veličina,
\item $x_1, x_2$ jsou hranice intervalu,
\item $\sum$ je součet pravděpodobností všech hodnot v intervalu,
\item $P(x)$ je pravděpodobnost, že $X = x$,
\item $F(x)$ je distribuční funkce.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Střední hodnota diskrétní náhodné veličiny}

Střední hodnota (očekávaná hodnota) diskrétní náhodné veličiny
je vážený průměr všech možných hodnot, kde váhami jsou jejich
pravděpodobnosti.

\begin{equation*}
E(X) = \sum_{x} x\,P(x)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $E(X)$ je očekávaná hodnota náhodné veličiny $X$,
\item $\sum$ je součet přes všechny možné hodnoty,
\item $x$ jsou jednotlivé možné hodnoty náhodné veličiny,
\item $P(x)=P(X=x)$ je pravděpodobnost, že $X$ nabude hodnoty $x$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Rozptyl diskrétní náhodné veličiny}

Rozptyl náhodné veličiny vyjadřuje míru variability
hodnot náhodné veličiny kolem její střední hodnoty.

\begin{equation*}
D(X) = E(X^2) - [E(X)]^2
     = \sum_x x^2 P(x) - \left(\sum_x x P(x)\right)^2
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $D(X)$ je rozptyl náhodné veličiny $X$,
\item $E(X)$ je očekávaná hodnota náhodné veličiny,
\item $E(X^2)$ je očekávaná hodnota druhé mocniny náhodné veličiny,
\item $\sum$ je součet přes všechny možné hodnoty,
\item $x$ jsou jednotlivé možné hodnoty náhodné veličiny,
\item $P(x)=P(X=x)$ je pravděpodobnost, že $X$ nabude hodnoty $x$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Hustota pravděpodobnosti spojité náhodné veličiny}

Hustota pravděpodobnosti vyjadřuje rychlost změny distribuční
funkce spojité náhodné veličiny.

\begin{equation*}
f(x) = F'(x)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $f(x)$ je hustota pravděpodobnosti,
\item $F(x)$ je distribuční funkce,
\item $F'(x)$ je derivace distribuční funkce podle $x$,
\item $x$ je hodnota náhodné veličiny.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Normalizační podmínka hustoty pravděpodobnosti}

Celková pravděpodobnost spojité náhodné veličiny
přes všechny možné hodnoty je rovna jedné.

\begin{equation*}
\int_{-\infty}^{\infty} f(x)\,dx = 1
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\int$ označuje integrál (součet přes všechny hodnoty),
\item $f(x)$ je hustota pravděpodobnosti,
\item $dx$ je integrační proměnná,
\item $-\infty, \infty$ vyjadřují integraci přes všechny možné hodnoty
náhodné veličiny.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Distribuční funkce spojité náhodné veličiny}

Distribuční funkce spojité náhodné veličiny udává
pravděpodobnost, že náhodná veličina $X$ bude menší
nebo rovna hodnotě $x$.

\begin{equation*}
F(x) = P(X \le x) = \int_{-\infty}^{x} f(t)\,dt
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $F(x)$ je distribuční funkce,
\item $P(X \le x)$ je pravděpodobnost, že náhodná veličina $X$ je menší nebo rovna $x$,
\item $\int$ označuje integrál,
\item $f(t)$ je hustota pravděpodobnosti,
\item $t$ je integrační proměnná,
\item $-\infty$ je dolní mez integrace,
\item $x$ je horní mez integrace.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Pravděpodobnost intervalu spojité náhodné veličiny}

Pravděpodobnost, že spojitá náhodná veličina $X$ nabývá hodnot
v intervalu $(x_1, x_2]$.

\begin{equation*}
P(x_1 < X \le x_2)
=
\int_{x_1}^{x_2} f(x)\,dx
=
F(x_2) - F(x_1)
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $P(x_1 < X \le x_2)$ je pravděpodobnost, že $X$ leží mezi $x_1$ a $x_2$,
\item $X$ je náhodná veličina,
\item $x_1, x_2$ jsou hranice intervalu,
\item $\int$ označuje integrál,
\item $f(x)$ je hustota pravděpodobnosti,
\item $dx$ je integrační proměnná,
\item $F(x)$ je distribuční funkce.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Střední hodnota spojité náhodné veličiny}

Střední hodnota (očekávaná hodnota) spojité náhodné veličiny
je vážený průměr všech možných hodnot, kde váhami jsou
hodnoty hustoty pravděpodobnosti.

\begin{equation*}
E(X) = \int_{-\infty}^{\infty} x f(x)\,dx
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $E(X)$ je očekávaná hodnota náhodné veličiny $X$,
\item $\int$ označuje integrál,
\item $x$ je hodnota náhodné veličiny,
\item $f(x)$ je hustota pravděpodobnosti,
\item $dx$ je integrační proměnná,
\item $-\infty, \infty$ vyjadřují integraci přes všechny možné hodnoty
náhodné veličiny.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Rozptyl spojité náhodné veličiny}

Rozptyl spojité náhodné veličiny vyjadřuje variabilitu
hodnot náhodné veličiny kolem její střední hodnoty.

\begin{equation*}
D(X)
=
\int_{-\infty}^{\infty} x^2 f(x)\,dx
-
\left[\int_{-\infty}^{\infty} x f(x)\,dx \right]^2
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $D(X)$ je rozptyl náhodné veličiny $X$,
\item $\int$ označuje integrál,
\item $x$ je hodnota náhodné veličiny,
\item $f(x)$ je hustota pravděpodobnosti,
\item $dx$ je integrační proměnná,
\item $\int x f(x)\,dx$ je očekávaná hodnota $E(X)$,
\item $\int x^2 f(x)\,dx$ je očekávaná hodnota $E(X^2)$.
\end{itemize}

%%-----------------------------------------------------------------------------
\subsubsection{Směrodatná odchylka náhodné veličiny}

Směrodatná odchylka vyjadřuje míru variability
hodnot náhodné veličiny kolem její střední hodnoty.

\begin{equation*}
\sigma = \sqrt{D(X)}
\end{equation*}

kde

\begin{itemize}[noitemsep]
\item $\sigma$ je směrodatná odchylka,
\item $D(X)$ je rozptyl náhodné veličiny $X$,
\item $\sqrt{\phantom{x}}$ označuje druhou odmocninu.
\end{itemize}

\subsection{Kombinatorika}
Mějme množinu $n$ prvků, ze které vybíráme $k$ prvků.
\begin{itemize}[noitemsep]
    \item \textbf{Permutace (Pořadí všech $n$ prvků):}
        Bez opakování: $P(n) = n!$ \\
        S opakováním: $P'_{k_1, \dots, k_m}(n) = \frac{n!}{k_1! \cdot \dots \cdot k_m!}$
    \item \textbf{Variace (Výběr $k$ prvků z $n$, záleží na pořadí):}
        Bez opakování: $V(k, n) = \frac{n!}{(n-k)!}$ \\
        S opakováním: $V'(k, n) = n^k$
    \item \textbf{Kombinace (Výběr $k$ prvků z $n$, nezáleží na pořadí):}
        Bez opakování: $K(k, n) = \binom{n}{k} = \frac{n!}{k!(n-k)!}$ \\
        S opakováním: $K'(k, n) = \binom{n+k-1}{k} = \frac{(n+k-1)!}{k!(n-1)!}$
\end{itemize}

\subsection{$L^p$ Normy (Vzdálenosti a penalizace)}
Obecně: $\|x\|_p = \left( \sum |x_i|^p \right)^{1/p}$. Klíčové pro metriky chyb a regularizaci.
\begin{itemize}[noitemsep]
    \item \textbf{$L^1$ (Manhattan):} $\|x\|_1 = \sum |x_i|$. Využití: Lasso regrese.
    \item \textbf{$L^2$ (Euklid):} $\|x\|_2 = \sqrt{\sum x_i^2}$. Využití: Ridge regrese (metoda nejmenších čtverců).
    \item \textbf{$L^\infty$ (Max):} $\|x\|_\infty = \max(|x_i|)$. Reprezentuje nejhorší možnou chybu.
\end{itemize}

\subsubsection*{Normální rozdělení (\texttt{norm})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Spojité rozdělení pro modelování např. času, délky či chyby měření.
    \item \textbf{Parametry:} \texttt{x} = zkoumaná hodnota (hranice), \texttt{mu} (\texttt{loc}) = střední hodnota, \texttt{sigma} (\texttt{scale}) = směrodatná odchylka, \texttt{q} = zadaná pravděpodobnost (pro hledání kvantilu).
\end{itemize}

\begin{funkcedict}
\texttt{norm.cdf(x, mu, sigma)} & \texttt{norm.cdf(16, loc=20, scale=4)} & \texttt{MÉNĚ NEŽ}: Šance, že hodnota bude maximálně $x$. \\
\texttt{norm.sf(x, mu, sigma)} & \texttt{norm.sf(20, loc=20, scale=4)} & \texttt{VÍCE NEŽ}: Ekvivalent \texttt{1 - cdf}. Šance, že hodnota překročí $x$. \\
\texttt{norm.cdf(b) - norm.cdf(a)} & \texttt{norm.cdf(28, 20, 4) - norm.cdf(12, 20, 4)} & \texttt{MEZI}: Šance, že se hodnota trefí do intervalu $(a, b)$. \\
\texttt{norm.ppf(q, mu, sigma)} & \texttt{norm.ppf(0.4, loc=20, scale=4)} & \texttt{Hledání kvantilu}: Najde hodnotu $x$, pro kterou platí $P(X < x) = q$. \\
\end{funkcedict}

\subsubsection*{Binomické rozdělení (\texttt{binom})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Diskrétní rozdělení. Počet úspěchů v pevně daném počtu nezávislých pokusů.
    \item \textbf{Aproximace:} Pokud je počet pokusů $n \ge 30$ a pravděpodobnost úspěchu $p \le 0.05$, blíží se Poissonovu rozdělení.
    \item \textbf{Parametry:} \texttt{k} = zkoumaný počet úspěchů, \texttt{n} = celkový počet pokusů, \texttt{p} = pravděpodobnost úspěchu.
\end{itemize}

\begin{funkcedict}
\texttt{binom.pmf(k, n, p)} & \texttt{binom.pmf(4, n=10, p=1/6)} & \texttt{Pravděpodobnost PŘESNĚ}: Šance, že úspěch nastane přesně $k$-krát. \\
\texttt{binom.cdf(k, n, p)} & \texttt{binom.cdf(2, n=10, p=1/6)} & \texttt{Pravděpodobnost NEJVÝŠE}: Šance, že úspěchů bude $0$ až $k$ (včetně). \\
\texttt{1 - binom.cdf(k, n, p)} & \texttt{1 - binom.cdf(2, n=10, p=1/6)} & \texttt{Pravděpodobnost VÍCE NEŽ}: Šance, že úspěchů bude striktně více než $k$. \\
\end{funkcedict}

\subsubsection*{Poissonovo rozdělení (\texttt{poisson})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Diskrétní rozdělení. Počet výskytů události za jednotku času nebo prostoru, když známe průměr.
    \item \textbf{Parametry:} \texttt{k} = zkoumaný počet událostí, \texttt{mu} = průměrný počet událostí ($\lambda$).
\end{itemize}

\begin{funkcedict}
\texttt{poisson.pmf(k, mu)} & \texttt{poisson.pmf(2, mu=0.4)} & \texttt{Pravděpodobnost PŘESNĚ}: Šance, že nastanou přesně $k$ události při průměru $\mu$. \\
\texttt{poisson.cdf(k, mu)} & \texttt{poisson.cdf(2, mu=0.4)} & \texttt{Pravděpodobnost NEJVÝŠE}: Šance, že událostí bude $0$ až $k$ (včetně). \\
\texttt{1 - poisson.cdf(k, mu)} & \texttt{1 - poisson.cdf(2, mu=0.4)} & \texttt{Pravděpodobnost VÍCE NEŽ}: Šance na více než $k$ událostí. \\
\end{funkcedict}

\subsubsection*{Hypergeometrické rozdělení (\texttt{hypergeom})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Diskrétní rozdělení. Výběr \textbf{bez vracení} z konečné populace (např. zmetky v zásilce).
    \item \textbf{Aproximace:} Pokud $N/M \le 0.05$, výběr bez vracení se chová téměř jako výběr s vracením (Binomické).
    \item \textbf{Parametry:} \texttt{k} = hledané prvky ve výběru, \texttt{M} = populace, \texttt{n} = hledané prvky v populaci, \texttt{N} = velikost výběru.
\end{itemize}

\begin{funkcedict}
\texttt{hypergeom.pmf(k, M, n, N)} & \texttt{hypergeom.pmf(0, 200, 10, 5)} & \texttt{Pravděpodobnost PŘESNĚ}: Šance na $k$ zmetků ve výběru $N$ z $M$, kde je celkem $n$ zmetků. \\
\end{funkcedict}

\subsubsection*{Studentovo t-rozdělení (\texttt{t})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Spojité rozdělení. Používá se pro konstrukci intervalů spolehlivosti a testování (neznáme-li rozptyl populace).
    \item \textbf{Parametry:} \texttt{df} = stupně volnosti, \texttt{q} = hladina (pravděpodobnost).
\end{itemize}

\begin{funkcedict}
\texttt{t.pdf(x, df)} & \texttt{t.pdf(x, df=15)} & \texttt{Hustota pravděpodobnosti}: Hodnota křivky pro bod $x$. \\
\texttt{t.ppf(q, df)} & \texttt{t.ppf(1 - 0.05/2, df=19)} & \texttt{Kritická hodnota}: Kvantil pro t-rozdělení (pro meze intervalu spolehlivosti). \\
\end{funkcedict}

\subsection{Intervaly spolehlivosti a velikost výběru}

% 1. Sekce pro průměr (střední hodnotu)
\subsubsection*{Interval pro průměr ($\mu$)}
\textbf{Kdy použít:} Pro odhad průměrné hodnoty měřitelného znaku (váha, čas, délka).
\begin{itemize}
    \item \textbf{Normální rozdělení (\texttt{norm}):} Pokud známe rozptyl populace $\sigma^2$ nebo je výběr velký ($n \ge 30$).
    \item \textbf{Studentovo t-rozdělení (\texttt{t}):} Pokud rozptyl populace neznáme a výběr je malý ($n < 30$).
    \item \textbf{Typy kvantilů:} Pro \textbf{oboustranný} interval používáme $1 - \alpha/2$, pro \textbf{jednostranný} (jen horní nebo dolní mez) používáme $1 - \alpha$.
\end{itemize}

\textbf{Parametry:} \texttt{x\_bar} (výběrový průměr), \texttt{s} (výběrová směr. odchylka), \texttt{n} (rozsah výběru), \texttt{alpha} ($\alpha$, hladina významnosti).

\begin{funkcedict}
\texttt{Kritická hodnota} & \texttt{crit = t.ppf(1 - alpha/2, df=n-1)} & Kvantil pro t-interval (neznámé $\sigma$). Pro velké $n$ lze nahradit \texttt{norm.ppf}. \\
\texttt{Mezní chyba (Mean)} & \texttt{err = crit * (s / np.sqrt(n))} & Výpočet odchylky pro průměr. \texttt{s} je výběrová směrodatná odchylka. \\
\texttt{Meze intervalu} & \texttt{lower, upper = p - err, p + err} & Skutečný parametr leží v intervalu s pravděpodobností $1 - \alpha$. \\
\end{funkcedict}

\vspace{3mm}

% 2. Sekce pro podíl (relativní četnost)
\subsubsection*{Interval pro podíl ($\pi$)}
\textbf{Kdy použít:} Pro odhad procentuálního zastoupení (např. podíl zmetků, preference voličů).
\begin{itemize}
    \item Vždy využíváme \textbf{normální rozdělení} (Z-kvantil).
    \item \textbf{Typy kvantilů:} Obdobně jako u průměru platí $1 - \alpha/2$ pro oboustranný a $1 - \alpha$ pro jednostranný odhad.
\end{itemize}

\textbf{Parametry:} \texttt{p} (výběrová relativní četnost), \texttt{n} (celkový počet testovaných jednotek).

\begin{funkcedict}
\texttt{Kritická hodnota} & \texttt{z\_crit = norm.ppf(1 - alpha / 2)} & Pro podíl se vždy používá normální rozdělení (Z-kvantil). \\
\texttt{Mezní chyba (Prop)} & \texttt{err = z\_crit * np.sqrt(p * (1-p) / n)} & Výpočet odchylky pro proporci $p$. \\
\texttt{Meze intervalu} & \texttt{lower, upper = p - err, p + err} & Skutečný parametr leží v intervalu s pravděpodobností $1 - \alpha$. \\
\end{funkcedict}

\vspace{3mm}

% 3. Výpočet potřebného rozsahu výběru
\subsubsection*{Potřebná velikost výběru ($n$)}
\textbf{Kdy použít:} Hledáme, kolik jednotek musíme vybrat ($n$), aby se náš odhad nemýlil o více než stanovenou chybu $\Delta$ (\texttt{delta}).
\begin{itemize}
    \item \textbf{Zaokrouhlování:} Protože $n$ musí být celé číslo, výsledek se \textbf{vždy zaokrouhluje nahoru} (funkce \texttt{math.ceil}), jinak by chyba byla větší než povolená.
\end{itemize}

\textbf{Parametry:} \texttt{delta} ($\Delta$, požadovaná přesnost), \texttt{sigma} (předpokládaná směr. odchylka).

\begin{funkcedict}
\texttt{Minimální n (Mean)} & \texttt{n = math.ceil((z\_crit * sigma / delta)**2)} & Pro průměr. $\Delta$ je cílová přesnost. Výsledek \textbf{vždy} zaokrouhli nahoru. \\
\texttt{Minimální n (Prop)} & \texttt{n = math.ceil(p*(1-p) * (z\_crit/delta)**2)} & Pro podíl. Používá se pro plánování reprezentativního výzkumu. \\
\end{funkcedict}

\subsection{Testování hypotéz o průměru ($\mu$)}

\textbf{Volba rozdělení:} Pro $n \ge 30$ využíváme \texttt{norm}, pro $n < 30$ využíváme \texttt{t} s \texttt{df = n-1}. \\
\textbf{Parametry:} \texttt{mu0} (testovaná hodnota), \texttt{x\_bar} (naměřený průměr), \texttt{s} (odchylka), \texttt{n} (rozsah).

\subsubsection*{1. Výpočet testového kritéria}
\texttt{u = (x\_bar - mu0) / (s / np.sqrt(n))}

\subsubsection*{2. Vyhodnocení testu ($p$-hodnota nebo kritická hodnota)}

\noindent \begin{tabular}{|l|l|l|}
\hline
\textbf{Typ testu} & \textbf{Alternativa $H_1$} & \textbf{Vzorec pro $p$-hodnotu (Python)} \\ \hline
Levostranný & $E(X) < \mu_0$ & \texttt{p = norm.cdf(u)} \\ \hline
Pravostranný & $E(X) > \mu_0$ & \texttt{p = 1 - norm.cdf(u)} \\ \hline
Oboustranný & $E(X) \neq \mu_0$ & \texttt{p = 2 * (1 - norm.cdf(abs(u)))} \\ \hline
\end{tabular} \vspace{12pt}

\noindent \textbf{Rozhodnutí pomocí $p$-hodnoty:} \vspace{4pt}

\noindent \begin{tabular}{|l|l|}
\hline
\textbf{Podmínka} & \textbf{Výsledek testu} \\ \hline
$p < \alpha$ & \textbf{ZAMÍTÁME $H_0$} (rozdíl je statisticky významný) \\ \hline
$p \ge \alpha$ & \textbf{NEZAMÍTÁME $H_0$} (rozdíl je pravděpodobně náhodný) \\ \hline
\end{tabular} \vspace{12pt}

\noindent \textbf{Rozhodnutí pomocí kritické hodnoty ($u_{\text{krit}}$):} \vspace{4pt}\\
\texttt{u\_krit = norm.ppf(1 - alpha)} \hspace{10pt} (u oboustranného testu \texttt{1 - alpha/2}) \vspace{8pt}

\noindent \begin{tabular}{|l|l|l|}
\hline
\textbf{Typ testu} & \textbf{Alternativa $H_1$} & \textbf{Zamítáme $H_0$, pokud platí:} \\ \hline
Levostranný & $E(X) < \mu_0$ & $u < -u_{\text{krit}}$ \\ \hline
Pravostranný & $E(X) > \mu_0$ & $u > u_{\text{krit}}$ \\ \hline
Oboustranný & $E(X) \neq \mu_0$ & $|u| \ge u_{\text{krit}}$ \\ \hline
\end{tabular} \vspace{8pt}

\noindent \hspace{15pt} \textit{Poznámka: U malého vzorku zaměňujeme \texttt{norm} za \texttt{t} s parametrem \texttt{df=n-1}.}

\subsubsection*{3. Automatické statistické testy (z pole dat)}
\noindent Využíváme funkce z knihovny \texttt{scipy.stats}, které provádějí kompletní výpočet přímo z naměřených hodnot.

\noindent \textbf{Vstupní parametry:}
\begin{itemize}
    \item \texttt{data, A, B, C}: Nezávislé vzorky měření (např. různé skupiny studentů nebo odrůd).
    \item \texttt{pred, po}: Závislá (párová) měření na stejných objektech (typicky testování „před a po“).
    \item \texttt{O, E}: Pozorované (\texttt{O}) a očekávané (\texttt{E}) četnosti výskytů.
    \item \texttt{table}: Kontingenční tabulka četností zadaná jako 2D pole (matice).
    \item \texttt{x, y}: Dvě sady proměnných pro zjištění síly jejich vzájemné závislosti.
\end{itemize}

\noindent \begin{funkcedict}
\texttt{Jednovýběrový t-test} & \texttt{stats.ttest\_1samp(data, popmean=mu0)} & Porovnání průměru vzorku s pevnou hodnotou $\mu_0$. \\
\texttt{Dvouvýběrový t-test} & \texttt{stats.ttest\_ind(A, B, equal\_var=False)} & Porovnání průměrů dvou nezávislých skupin (Welchův test). \\
\texttt{Párový t-test} & \texttt{stats.ttest\_rel(po, pred)} & Porovnání stejné skupiny ve dvou různých situacích. \\
\texttt{ANOVA (Analýza rozptylu)} & \texttt{stats.f\_oneway(A, B, C)} & Testování shody průměrů u tří a více skupin. \\
\texttt{Chí-kvadrát shoda} & \texttt{stats.chisquare(O, f\_exp=E)} & Test dobré shody (např. spravedlivost kostky nebo náhodnost). \\
\texttt{Chí-kvadrát nezávislost} & \texttt{stats.chi2\_contingency(table)} & Test závislosti kategoriálních znaků z kontingenční tabulky. \\
\texttt{Pearsonova korelace} & \texttt{stats.pearsonr(x, y)} & Měření síly lineární závislosti dvou spojitých proměnných. \\
\texttt{Spearmanova korelace} & \texttt{stats.spearmanr(x, y)} & Měření síly monotónní závislosti (i pro nelineární vztahy). \\
\end{funkcedict}

\noindent \textbf{Návratové hodnoty a rozhodnutí:}
\begin{itemize}
    \item \texttt{res.pvalue}: Klíčová hodnota pro rozhodnutí. Pokud \textbf{$p < \alpha$}, nulovou hypotézu $H_0$ \textbf{zamítáme}.
    \item \texttt{res.statistic}: Hodnota testového kritéria (čím je vyšší, tím více se data liší od hypotézy).
    \item \textit{Výjimka:} \texttt{chi2\_contingency} vrací výsledky jako n-tici \texttt{(statistika, p-hodnota, df, expected\_table)}.
\end{itemize}

\section{Práce s daty v R (Cheatsheet)}

\subsection{Náhodné veličiny a rozdělení}
\texttt{d}xxx = hustota pravděpodobnosti, \texttt{p}xxx = distribuční funkce $F(x)$, \texttt{q}xxx = kvantil, \texttt{r}xxx = náhodné číslo.

\subsubsection*{Normální rozdělení (\texttt{norm})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Spojité rozdělení pro modelování např. výšky, chyby měření či doby čekání.
    \item \textbf{Parametry:} \texttt{mean} ($\mu$) = střední hodnota, \texttt{sd} ($\sigma$) = směrodatná odchylka.
\end{itemize}

\begin{funkcedict}
\texttt{pnorm(x, mean, sd)} & \texttt{pnorm(16, mean=20, sd=4)} & \texttt{MÉNĚ NEŽ}: $P(X \le x)$. \\
\texttt{1 - pnorm(x, mean, sd)} & \texttt{1 - pnorm(20, mean=20, sd=4)} & \texttt{VÍCE NEŽ}: $P(X > x)$. \\
\texttt{pnorm(b,m,s) - pnorm(a,m,s)} & \texttt{pnorm(28,20,4) - pnorm(12,20,4)} & \texttt{MEZI}: $P(a < X \le b)$. \\
\texttt{qnorm(q, mean, sd)} & \texttt{qnorm(0.975, mean=0, sd=1)} & Kvantil: najde $x$ pro $P(X \le x) = q$. \\
\end{funkcedict}

\subsubsection*{Binomické rozdělení (\texttt{binom})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Diskrétní rozdělení. Počet úspěchů v pevně daném počtu nezávislých pokusů.
    \item \textbf{Aproximace:} Pokud je počet pokusů $n \ge 30$ a pravděpodobnost úspěchu $p \le 0{,}05$, blíží se Poissonovu rozdělení ($\lambda = np$).
    \item \textbf{Parametry:} \texttt{size} ($n$) = celkový počet pokusů, \texttt{prob} ($p$) = pravděpodobnost úspěchu.
\end{itemize}

\begin{funkcedict}
\texttt{dbinom(k, size, prob)} & \texttt{dbinom(4, size=10, prob=1/6)} & \texttt{PŘESNĚ}: $P(X = k)$. \\
\texttt{pbinom(k, size, prob)} & \texttt{pbinom(2, size=10, prob=1/6)} & \texttt{NEJVÝŠE}: $P(X \le k)$. \\
\texttt{1 - pbinom(k, size, prob)} & \texttt{1 - pbinom(2, size=10, prob=1/6)} & \texttt{VÍCE NEŽ}: $P(X > k)$. \\
\end{funkcedict}

\subsubsection*{Poissonovo rozdělení (\texttt{pois})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Diskrétní rozdělení. Počet výskytů události za jednotku času nebo prostoru, když známe průměr.
    \item \textbf{Parametry:} \texttt{lambda} ($\lambda$) = průměrný počet událostí za interval.
\end{itemize}

\begin{funkcedict}
\texttt{dpois(k, lambda)} & \texttt{dpois(2, lambda=0.4)} & \texttt{PŘESNĚ}: $P(X = k)$. \\
\texttt{ppois(k, lambda)} & \texttt{ppois(2, lambda=0.4)} & \texttt{NEJVÝŠE}: $P(X \le k)$. \\
\texttt{1 - ppois(k, lambda)} & \texttt{1 - ppois(2, lambda=0.4)} & \texttt{VÍCE NEŽ}: $P(X > k)$. \\
\end{funkcedict}

\subsubsection*{Hypergeometrické rozdělení (\texttt{hyper})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Diskrétní rozdělení. Výběr \textbf{bez vracení} z konečné populace (např. zmetky v zásilce).
    \item \textbf{Aproximace:} Pokud $\frac{k}{m+n} \le 0{,}05$, výběr bez vracení se chová téměř jako výběr s vracením (Binomické).
    \item \textbf{Parametry:} \texttt{m} = počet hledaných prvků v populaci, \texttt{n} = počet ostatních prvků v populaci, \texttt{k} = velikost výběru.
\end{itemize}

\begin{funkcedict}
\texttt{dhyper(x, m, n, k)} & \texttt{dhyper(0, m=10, n=190, k=5)} & \texttt{PŘESNĚ}: $P(X = x)$, zde: 0 zmetků z populace 200 (10 zmetků), výběr 5. \\
\texttt{phyper(x, m, n, k)} & \texttt{phyper(2, m=10, n=190, k=5)} & \texttt{NEJVÝŠE}: $P(X \le x)$. \\
\end{funkcedict}

\subsubsection*{Studentovo t-rozdělení (\texttt{t})}
\begin{itemize}[label={}, leftmargin=1.5em, nosep]
    \item Spojité rozdělení. Používá se pro intervaly spolehlivosti a testování, pokud neznáme rozptyl populace.
    \item \textbf{Parametry:} \texttt{df} = stupně volnosti ($n - 1$ pro jednovýběrový t-test).
\end{itemize}

\begin{funkcedict}
\texttt{pt(x, df)} & \texttt{pt(-1.96, df=19)} & Distribuční funkce $F(x) = P(T \le x)$. \\
\texttt{qt(q, df)} & \texttt{qt(0.975, df=19)} & Kvantil: kritická hodnota pro t-test nebo IS. \\
\end{funkcedict}

\subsubsection*{Chí-kvadrát rozdělení (\texttt{chisq}) a F-rozdělení (\texttt{f})}

\begin{funkcedict}
\texttt{pchisq(x, df)} & \texttt{pchisq(7.81, df=3)} & $P(\chi^2 \le x)$ -- distribuční funkce. \\
\texttt{qchisq(q, df)} & \texttt{qchisq(0.99, df=3)} & Kritická hodnota pro $\chi^2$ test. \\
\texttt{pf(x, df1, df2)} & \texttt{pf(3.86, df1=2, df2=9)} & $P(F \le x)$ -- distribuční funkce pro ANOVA. \\
\texttt{qf(q, df1, df2)} & \texttt{qf(0.95, df1=2, df2=9)} & Kritická hodnota pro F-test (ANOVA). \\
\end{funkcedict}

\subsection{Intervaly spolehlivosti a velikost výběru}

\subsubsection*{Interval pro průměr ($\mu$)}
\textbf{Kdy použít:} Pro odhad průměrné hodnoty měřitelného znaku (váha, čas, délka).
\begin{itemize}
    \item \textbf{Normální rozdělení:} Pokud známe $\sigma^2$ nebo je výběr velký ($n \ge 30$): použijeme \texttt{qnorm}.
    \item \textbf{t-rozdělení:} Neznámý $\sigma^2$ nebo malý výběr ($n < 30$): použijeme \texttt{qt} s \texttt{df = n-1}.
    \item \textbf{Jednostranný IS:} Kvantil $1 - \alpha$ místo $1 - \alpha/2$.
\end{itemize}

\textbf{Parametry:} \texttt{x} (vektor dat), \texttt{xbar} (průměr), \texttt{s} (výběrová směr. odchylka), \texttt{n} (rozsah), \texttt{alpha} ($\alpha$).

\begin{funkcedict}
\texttt{t.test(x)\$conf.int} &
\begin{lstlisting}
x <- c(163,175,177,165,171,174)
t.test(x, conf.level = 0.95)$conf.int
\end{lstlisting}
& Vrací 95\,\% IS pro $\mu$ (neznámý $\sigma^2$). \\

\texttt{Ručně (t-rozdělení)} &
\begin{lstlisting}
xbar <- mean(x); s <- sd(x)
n <- length(x); alpha <- 0.05
t_kr <- qt(1 - alpha/2, df = n - 1)
c(xbar - t_kr*s/sqrt(n),
  xbar + t_kr*s/sqrt(n))
\end{lstlisting}
& Ruční výpočet IS s t-rozdělením. \\

\texttt{Ručně (Z, známý $\sigma$)} &
\begin{lstlisting}
sigma <- 7
z_kr <- qnorm(1 - alpha/2)
c(xbar - z_kr*sigma/sqrt(n),
  xbar + z_kr*sigma/sqrt(n))
\end{lstlisting}
& Ruční výpočet IS s normálním rozdělením (Z-kvantil). \\
\end{funkcedict}

\vspace{3mm}

\subsubsection*{Interval pro podíl ($\pi$)}
\textbf{Kdy použít:} Pro odhad procentuálního zastoupení (např. podíl zmetků, preference voličů).
\begin{itemize}
    \item Vždy využíváme \textbf{normální rozdělení} (Z-kvantil).
    \item \textbf{Typy kvantilů:} Oboustranný: $1 - \alpha/2$, jednostranný: $1 - \alpha$.
\end{itemize}

\textbf{Parametry:} \texttt{x} (počet úspěchů), \texttt{n} (celkový počet), \texttt{p} = \texttt{x/n}.

\begin{funkcedict}
\texttt{prop.test(x, n)\$conf.int} &
\begin{lstlisting}
prop.test(x=110, n=200,
  conf.level=0.95)$conf.int
\end{lstlisting}
& Wilsonův IS pro proporci (vestavěná funkce). \\

\texttt{Ručně (Waldův IS)} &
\begin{lstlisting}
x_vol <- 110; n_vol <- 200
p <- x_vol / n_vol
z_kr <- qnorm(0.975)
se <- sqrt(p*(1-p)/n_vol)
c(p - z_kr*se, p + z_kr*se)
\end{lstlisting}
& Waldův IS -- jednoduchá normální aproximace. \\
\end{funkcedict}

\vspace{3mm}

\subsubsection*{Potřebná velikost výběru ($n$)}
\textbf{Kdy použít:} Hledáme, kolik jednotek musíme vybrat ($n$), aby se náš odhad nemýlil o více než stanovenou chybu $\Delta$.
\begin{itemize}
    \item Výsledek \textbf{vždy zaokrouhlit nahoru} (funkce \texttt{ceiling()}), jinak by chyba byla větší než povolená.
\end{itemize}

\textbf{Parametry:} \texttt{delta} ($\Delta$, požadovaná přesnost), \texttt{sigma} (předpokládaná směr. odchylka).

\begin{funkcedict}
\texttt{Minimální n (průměr)} &
\begin{lstlisting}
alpha <- 0.05; sigma <- 10
delta <- 2  # pozadovana presnost
z_kr <- qnorm(1 - alpha/2)
ceiling((z_kr * sigma / delta)^2)
\end{lstlisting}
& Pro průměr. $\Delta$ je cílová přesnost. Výsledek \textbf{vždy} zaokrouhli nahoru. \\

\texttt{Minimální n (podíl)} &
\begin{lstlisting}
p <- 0.5; delta <- 0.05  # +-5 p.b.
ceiling(p*(1-p)*(z_kr/delta)^2)  # 385
\end{lstlisting}
& Pro podíl. Používá se pro plánování reprezentativního výzkumu. \\
\end{funkcedict}

\subsection{Testování hypotéz}

\textbf{Volba rozdělení:} Pro $n \ge 30$ lze použít \texttt{qnorm}/\texttt{pnorm}, pro $n < 30$ použijeme \texttt{qt}/\texttt{pt} s \texttt{df = n-1}. \\
\textbf{Parametry:} \texttt{mu0} (testovaná hodnota), \texttt{xbar} (výběrový průměr), \texttt{s} (výběrová směr. odchylka), \texttt{n} (rozsah), \texttt{alpha} ($\alpha$).

\subsubsection*{1. Jednovýběrový t-test (střední hodnota)}

\texttt{T = (xbar - mu0) / (s / sqrt(n))} \quad $\sim t(n-1)$

\vspace{4pt}
\noindent \begin{tabular}{|l|l|l|}
\hline
\textbf{Typ testu} & \textbf{Alternativa $H_1$} & \textbf{p-hodnota v R} \\ \hline
Levostranný & $\mu < \mu_0$ & \texttt{pt(T, df=n-1)} \\ \hline
Pravostranný & $\mu > \mu_0$ & \texttt{1 - pt(T, df=n-1)} \\ \hline
Oboustranný & $\mu \ne \mu_0$ & \texttt{2*(1 - pt(abs(T), df=n-1))} \\ \hline
\end{tabular} \vspace{8pt}

\begin{funkcedict}
\texttt{t.test(x, mu=mu0, alternative=...)} &
\begin{lstlisting}
t.test(x, mu = 25000,
       alternative = "less")
# "greater" / "two.sided"
\end{lstlisting}
& Automatický t-test z vektoru dat. \\

\texttt{Ručně} &
\begin{lstlisting}
n <- 49; xbar <- 23500; alpha <- 0.05
mu0 <- 25000; s <- 5500
T <- (xbar - mu0)/(s/sqrt(n))
p_val <- pt(T, df = n - 1)
t_kr  <- qt(alpha, df = n - 1)
\end{lstlisting}
& Ruční výpočet (jednostranný levostranný). \\
\end{funkcedict}

\subsubsection*{2. Párový t-test}

\begin{funkcedict}
\texttt{t.test(pred, po, paired=TRUE)} &
\begin{lstlisting}
t.test(pred, po,
       paired = TRUE,
       alternative = "two.sided")
\end{lstlisting}
& Porovnání stejné skupiny ve dvou různých situacích (např. před a po zásahu). \\
\end{funkcedict}

\subsubsection*{3. Test proporce}

$Z = \dfrac{\hat{p} - \pi_0}{\sqrt{\pi_0(1-\pi_0)/n}} \sim N(0,1)$

\begin{funkcedict}
\texttt{prop.test(x, n, p=p0, ...)} &
\begin{lstlisting}
prop.test(x=90, n=150, p=0.5,
  alternative="two.sided",
  correct=FALSE)
\end{lstlisting}
& Bez Yatesovy korekce odpovídá ručnímu Z-testu. \\

\texttt{Ručně} &
\begin{lstlisting}
p0 <- 0.5; n <- 150; orel <- 90
phat <- orel/n
Z <- (phat-p0)/sqrt(p0*(1-p0)/n)
z_kr <- qnorm(1 - 0.01/2)
p_val <- 2*(1 - pnorm(abs(Z)))
\end{lstlisting}
& Oboustranný Z-test symetrie mince ($\alpha = 0{,}01$). \\
\end{funkcedict}

\subsubsection*{4. Dvouvýběrový t-test (Welchův)}

$T = \dfrac{\bar{x}_1 - \bar{x}_2}{\sqrt{s_1^2/n_1 + s_2^2/n_2}}$

\begin{funkcedict}
\texttt{t.test(x1, x2, var.equal=FALSE)} &
\begin{lstlisting}
t.test(x1, x2,
       var.equal = FALSE,
       alternative = "two.sided")
\end{lstlisting}
& Welchův test (různé rozptyly). \\

\texttt{Ručně (ze souhrnných statistik)} &
\begin{lstlisting}
n1<-38; x1<-5.3875; s1_2<-0.2698
n2<-35; x2<-4.7;    s2_2<-0.24
T <- (x1-x2)/sqrt(s1_2/n1+s2_2/n2)
df_w <- (s1_2/n1+s2_2/n2)^2 /
  ((s1_2/n1)^2/(n1-1)+
   (s2_2/n2)^2/(n2-1))
p_val <- 2*(1-pt(abs(T), df=df_w))
\end{lstlisting}
& Welch-Satterthwaiteovy stupně volnosti ($s_1^2$, $s_2^2$ jsou rozptyly). \\
\end{funkcedict}

cor.test(x, y,

& Pearsonova korelace -- měří sílu lineární závislosti dvou spojitých proměnných. \\

\texttt{cor.test(x, y, method=...)} &

\subsection{ANOVA (Jednocestná analýza rozptylu)}

\textbf{H$_0$:} $\mu_1 = \mu_2 = \dots = \mu_k$, \textbf{H$_1$:} alespoň dva průměry se liší. \\
Test je vždy \textbf{pravostranný}: zamítáme $H_0$, pokud $F \ge F_{1-\alpha}(k-1, n-k)$.

$$F = \frac{S_m/(k-1)}{S_v/(n-k)} \sim F(k-1, n-k)$$

\begin{funkcedict}
\texttt{aov(y \textasciitilde\ skupina)} &
\begin{lstlisting}
y <- c(91,81,74,57, 83,72,63,47,
       71,69,58,40)
skupina <- factor(rep(c("A","B","C"),
                      each=4))
m <- aov(y ~ skupina)
summary(m)
\end{lstlisting}
& Automatická ANOVA s tabulkou výsledků. \\

\texttt{Skupinové průměry} &
\begin{lstlisting}
tapply(y, skupina, mean)
\end{lstlisting}
& Průměry v každé skupině. \\

\texttt{Kritická hodnota F} &
\begin{lstlisting}
k <- nlevels(skupina); n <- length(y)
qf(0.95, df1 = k-1, df2 = n-k)
\end{lstlisting}
& Kvantil F-rozdělení pro $\alpha = 0{,}05$. \\

\texttt{Ručně (SS)} &
\begin{lstlisting}
k <- nlevels(skupina); n <- length(y)
grand <- mean(y)
SS_m <- sum(tapply(y,skupina,
  function(g) length(g)*(mean(g)-grand)^2))
SS_v <- sum(tapply(y,skupina,
  function(g) sum((g-mean(g))^2)))
F_stat <- (SS_m/(k-1))/(SS_v/(n-k))
\end{lstlisting}
& Ruční výpočet součtů čtverců mezi skupinami a uvnitř skupin. \\
\end{funkcedict}

\subsection{Chí-kvadrát testy}

\subsubsection*{Test dobré shody}
\textbf{H$_0$:} výběr odpovídá předpokládanému rozdělení. \\
$G = \sum_j \dfrac{(n_j - e_j)^2}{e_j} \sim \chi^2(k-1)$, kde $e_j = n \cdot \pi_{0,j}$.

\begin{funkcedict}
\texttt{chisq.test(emp, p=p\_teor)} &
\begin{lstlisting}
emp <- c(A=82, B=38, AB=26, O=69)
p0  <- c(A=0.41,B=0.14,AB=0.07,O=0.38)
chisq.test(emp, p = p0)
\end{lstlisting}
& Automatický test dobré shody. Vektor \texttt{p0} musí mít součet 1. \\

\texttt{Ručně} &
\begin{lstlisting}
teor <- sum(emp) * p0
G <- sum((emp - teor)^2 / teor)
p_val <- 1 - pchisq(G, df=length(emp)-1)
chi_kr <- qchisq(0.99, df=length(emp)-1)
\end{lstlisting}
& Ruční výpočet testové statistiky a p-hodnoty. \\
\end{funkcedict}

\subsubsection*{Test nezávislosti (kontingenční tabulka)}
\textbf{H$_0$:} řádkový a sloupcový znak jsou nezávislé. \\
$e_{ij} = \dfrac{n_{i\bullet}\, n_{\bullet j}}{n}$, \quad $G \sim \chi^2\bigl((r-1)(s-1)\bigr)$.

\begin{funkcedict}
\texttt{chisq.test(tabulka)} &
\begin{lstlisting}
kt <- matrix(c(1768,807,189,47,
               946,1387,746,53,
               115,438,288,16),
             nrow=3, byrow=TRUE)
res <- chisq.test(kt)
res$p.value      # p-hodnota
res$expected     # teoreticke cetnosti
\end{lstlisting}
& Test nezávislosti z kontingenční tabulky. \\

\texttt{Cramérovo V} &
\begin{lstlisting}
G <- as.numeric(res$statistic)
n_celk <- sum(kt)
r <- nrow(kt); s_kt <- ncol(kt)
V <- sqrt(G / (n_celk*(min(r,s_kt)-1)))
C <- sqrt(G / (G + n_celk))
\end{lstlisting}
& Síla závislosti: $V \in [0,1]$. Pearsonův $C$ závisí na rozměrech tabulky. \\
\end{funkcedict}

\noindent \textbf{Interpretace Cramérova V} (orientační, $df^* = \min(r,s)-1$):
$< 0{,}1$ zanedbatelná $\cdot$ $0{,}1$--$0{,}3$ slabá $\cdot$ $0{,}3$--$0{,}5$ středně silná $\cdot$ $> 0{,}5$ silná závislost.

\vspace{4pt}
\noindent \textbf{Rozhodnutí pomocí p-hodnoty:}

\noindent \begin{tabular}{|l|l|}
\hline
\textbf{Podmínka} & \textbf{Výsledek testu} \\ \hline
$p < \alpha$ & \textbf{ZAMÍTÁME $H_0$} (rozdíl je statisticky významný) \\ \hline
$p \ge \alpha$ & \textbf{NEZAMÍTÁME $H_0$} (rozdíl je pravděpodobně náhodný) \\ \hline
\end{tabular}

\subsection{Korelační analýza}

\subsubsection*{Pearsonův a Spearmanův korelační koeficient}
\textbf{H$_0$:} $\rho = 0$ (žádná lineární/monotónní závislost). Testové kritérium $T = r\sqrt{(n-2)/(1-r^2)} \sim t(n-2)$.

\begin{funkcedict}
\texttt{cor(x, y)} &
\begin{lstlisting}
x <- c(1,2,0,5,3,4,2,6,1,3)
y <- c(0,1,0,4,2,3,1,5,0,2)
cor(x, y, method="pearson")   # linearni
cor(x, y, method="spearman")  # poradovy
\end{lstlisting}
